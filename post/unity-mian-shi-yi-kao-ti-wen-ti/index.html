array = {“Lua","C#","Java"}
for k,v in pairs(array) do
print(k,v)
end
//遍历array中内容

array[2]=nil
for k,v in ipairs(array) do//ipair:按照索引从1开始，递增遍历，遇到nil时即停止
print(k,v)
end
//只输出“Lua”
